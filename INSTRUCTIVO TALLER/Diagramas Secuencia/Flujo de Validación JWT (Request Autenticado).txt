    JwtAuthenticationFilter->>JwtService: extractUsername(token)
    JwtService-->>JwtAuthenticationFilter: "user-{id}"
    
    JwtAuthenticationFilter->>JwtAuthenticationFilter: extractUserIdFromSubject("user-{id}")

    UserRepository-->>JwtAuthenticationFilter: User
    
    JwtAuthenticationFilter->>TokenRepository: findByToken(jwt)
    TokenRepository-->>JwtAuthenticationFilter: Token
    
    alt Token revocado o expirado en DB
        JwtAuthenticationFilter->>MetricsService: incrementJwtExpired()
        JwtAuthenticationFilter-->>Client: 401 Unauthorized
    else Token válido en DB
        JwtAuthenticationFilter->>JwtService: isTokenValid(token, user)
        
        alt Firma o expiración inválida
            JwtService-->>JwtAuthenticationFilter: false
            JwtAuthenticationFilter->>MetricsService: incrementJwtInvalid()
            JwtAuthenticationFilter-->>Client: 401 Unauthorized
        else Token JWT válido
            JwtService-->>JwtAuthenticationFilter: true
            JwtAuthenticationFilter->>MetricsService: incrementJwtValidated()
            
            JwtAuthenticationFilter->>JwtAuthenticationFilter: validateRequiredClaims(jwt)
            JwtAuthenticationFilter->>JwtService: validateRoles(jwt, requiredRoles)
            JwtAuthenticationFilter->>JwtService: validateScopes(jwt, requiredScopes)
            
            JwtAuthenticationFilter->>SecurityContext: setAuthentication(authToken)
            JwtAuthenticationFilter->>Controller: doFilter(request, response)
            Controller-->>Client: 200 OK {response}
